			\section*{Probability Notes}
			\paragraph{} \ \\
			\begin{flalign*}
				\Ex{X} &= \mu = \sum_{x_i\in\Omega}x_i\P{x_i}&&\\
				\Ex{g(X)} &= \sum_{x_i\in\Omega}g(x)\,\P{x_i}&&\\
				\Ex{aX+b\,g(X)+c} &= a\Ex{X} + b\Ex{g(X)} + c&&\\
				\Var{X} &= \sigma^2 = \Ex{(X-\mu)^2}&&\\
				\Var{aX+b} &= a^2\Var{X}&&\\
				\Var{X} &= \Ex{X^2} - \mu^2&&
			\end{flalign*}
			\paragraph{Common Probability Distributions} \ \\
			\begin{itemize}
				\item Binomial Distribution:
				      \[
					      X \sim \text{Bin}(n,p) \Rightarrow
					      \begin{cases}
						      \P(X=x) = \binom{n}{x} p^x (1-p)^{n-x} \\
						      \Ex{X} = np                            \\
						      \Var{X} = np(1-p)
					      \end{cases}
				      \]
				\item Poisson Distribution:
				      \[
					      X \sim \text{Poi}(\lambda) \Rightarrow
					      \begin{cases}
						      \P(X=x) = e^{-\lambda} \frac{\lambda^x}{x!} \\
						      \Ex{X} = \lambda                            \\
						      \Var{X} = \lambda
					      \end{cases}
				      \]
				\item Geometric Distribution:
				      \[
					      X \sim \text{Geo}(p) \Rightarrow
					      \begin{cases}
						      \P(X=x) = p(1-p)^{x-1} \\
						      \Ex{X} = \frac{1}{p}   \\
						      \Var{X} = \frac{1-p}{p^2}
					      \end{cases}
				      \]
			\end{itemize}
		\end{minipage}
	}

			\paragraph{Transforms} \ \\
			\begin{itemize}
				\item MGF:
				      \begin{flalign*}
					      M(s) &= \Ex{e^{sX}}&&\\
					      \Ex{X^n} &= M^{(n)}(0)&&\\
					      \Var{X} &= M''(0) - M'(0)^2 &&
				      \end{flalign*}
				\item PGF:
				      \begin{flalign*}
					      G(z) &= \Ex{z^X}&&\\
					      \Ex{X} &= G'(1)&&\\
					      \Var{X} &= G''(1) + G'(1) - G'(1)^2 &&
				      \end{flalign*}
			\end{itemize}

			\paragraph{Conditional Probability} \ \\
			\[
				\P_{Y|x}(y) = \frac{\P_{XY}(x,y)}{\P_X(x)}
			\]
